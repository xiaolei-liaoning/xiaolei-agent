export const meta = {
  name: "热搜分析",
  description: "搜索百度热搜并分析趋势",

export default async function() {
  phase("搜索")
  const hotSearch = await agent("搜索百度热搜top10", {
    label: "热搜搜索", timeout: 60,
  })

  phase("分析")
  const analysis = await agent("分析热搜趋势", {
    label: "趋势分析", timeout: 120,
  })

  phase("生成报告")
  return await agent("生成分析报告并保存到桌面", {
    label: "报告生成", timeout: 120,
  })
}
